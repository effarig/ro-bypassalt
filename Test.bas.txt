alt% = 2
lalt% = 5
ralt% = 8

m% = 20

DIM c%(m%,6)

c%(0,0)=121:c%(0,1)=alt%  EOR &80:c%(0,2)=255
c%(1,0)=121:c%(1,1)=lalt% EOR &80:c%(1,2)=255
c%(2,0)=121:c%(2,1)=ralt% EOR &80:c%(2,2)=255
c%(3,0)=129:c%(3,1)=alt%  EOR &FF:c%(3,2)=255
c%(4,0)=129:c%(4,1)=lalt% EOR &FF:c%(4,2)=255
c%(5,0)=129:c%(5,1)=ralt% EOR &FF:c%(5,2)=255

c%(6,0)=121:c%(6,1)=0:c%(6,2)=0
c%(7,0)=129:c%(7,1)=1 EOR &7F:c%(7,2)=255
c%(8,0)=-1

REPEAT
  n% = 0
  g% = FALSE
  WHILE c%(n%,0)<>-1
    SYS"OS_Byte",c%(n%,0),c%(n%,1),c%(n%,2) TO c3%,c4%,c5%
    IF (c3%<>c%(n%,3)) OR (c4%<>c%(n%,4)) OR (c5%<>c%(n%,5)) THEN
      PRINT "";n%;": ";c%(n%,0);", ";c%(n%,1);" ";c%(n%,2);": ";c%(n%,3);"->";c3%;", ";c%(n%,4);"->";c4%;", ";~c%(n%,5);"->";~c5%
      c%(n%,3)=c3%
      c%(n%,4)=c4%
      c%(n%,5)=c5%
      g% = TRUE
    ENDIF
    n%+=1
  ENDWHILE
  IF g% THEN PRINT
  WAIT
UNTIL 0
